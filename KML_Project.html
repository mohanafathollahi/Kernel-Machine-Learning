#This function first unzip the *.gzip format, then convert each number in the file to a uin8 and return a 2D matrix or the image
base_dir = os.getcwd()
def extract_images(img_filename):
  with gzip.open(img_filename, 'r') as f:
        magic_number = int.from_bytes(f.read(4), 'big')
        image_count = int.from_bytes(f.read(4), 'big')
        row_count = int.from_bytes(f.read(4), 'big')
        column_count = int.from_bytes(f.read(4), 'big')
        print(f'row_count: {row_count} column_count: {column_count}')
        image_data = f.read()
        images = np.frombuffer(image_data, dtype=np.uint8)\
             .reshape((image_count, row_count, column_count))
        return images
    
def extract_labels(label_filename):
     with gzip.open(label_filename, 'r') as f:
         magic_number = int.from_bytes(f.read(4), 'big')
         label_count = int.from_bytes(f.read(4), 'big')
         label_data = f.read()
         labels = np.frombuffer(label_data, dtype=np.uint8)
         return labels

def linear_kernel(x, y):
    xx = np.dot(x, x)
    xy = np.dot(x, y)
    yy = np.dot(y, y)
    return math.sqrt(xx - (2 * xy) + yy)
In [ ]:
def polynomial_kernel(x, y, d = 3):
    xx = pow(1 + np.dot(x, x), d)
    xy = pow(1 + np.dot(x, y), d)
    yy = pow(1 + np.dot(y, y), d)
    return math.sqrt(xx - (2 * xy) + yy)
In [ ]:
def power_kernel(X, Y, d):
    return -1*euclidean_distances(X, Y)**d
In [ ]:
def log_kernel(X, Y,d):
    K = euclidean_distances(X, Y)**d+1
    np.log(K, K)      # imlicite in place, to save memory
    return -K

def confusion(true, pred):
    true.name = 'target'
    pred.name = 'predicted'
    cm = pd.crosstab(true.reset_index(drop=True), pred.reset_index(drop=True),normalize='index').round(4)*100
    cm = cm[cm.index]
    return cm

def kNN(X_train, y_train, X_test, y_test, K_MAX, dist_metrics):
    
    # find optimal k and metric for given data set
    scores = [[0] * len(dist_metrics) for i in range(math.ceil(K_MAX/2))]
    
    # print out used metrics
    if (isinstance(dist_metrics[0], str)):
        print(f'Metrics: {dist_metrics}')
    else:
        print(f'Metrics: {[d.__name__ for d in dist_metrics]}')

    # try out all different k and metrics
    for k in range(1, K_MAX+1, 2):
        for j in range(len(dist_metrics)):
            knn = neighbors.KNeighborsClassifier(n_neighbors=k, metric=dist_metrics[j])
            knn.fit(X_train, y_train)            
            scores[math.floor(k/2)][j] = knn.score(X_test, y_test)
        print(f'accuracies for k={k} : {scores[math.floor(k/2)]}') # print to check how far program is

    # select k and metric with highest accuracy
    scores_np = np.array(scores)
    optimal_index = np.unravel_index(np.argmax(scores_np), scores_np.shape)
    optimal_k = optimal_index[0] * 2 + 1
    optimal_metric_index = optimal_index[1]
    optimal_metric = dist_metrics[optimal_metric_index]
    print('\nInformation on optimal kNN classifier:')
    print(f'Optimal score: {scores[optimal_index[0]][optimal_index[1]]}')
    if (isinstance(optimal_metric, str)):
        print(f'Optimal metric: {optimal_metric}')
    else:
        print(f'Optimal metric: {optimal_metric.__name__}')
    print(f'Optimal k: {optimal_k}')
    
    # plot accuracies in graph
    for i in range(0,len(scores[0])):
        lw = (len(scores[0]) - i*0.5)
        ls = ['--',':','-.'][i%3]
        if (isinstance(dist_metrics[i], str)):
            plt.plot(range(1, K_MAX+1, 2), [row[i] for row in scores],
                     label = dist_metrics[i], linestyle = ls, linewidth = lw)
        else:
            plt.plot(range(1, K_MAX+1, 2), [row[i] for row in scores],
                     label = dist_metrics[i].__name__, linestyle = ls, linewidth = lw)
    plt.xlabel('k')
    plt.ylabel('Accuracy')
    plt.legend()
    plt.show()

    # predict using optimal metric and k
    knn_optimal = neighbors.KNeighborsClassifier(n_neighbors=optimal_k, metric = optimal_metric)
    knn_optimal.fit(X_train, y_train)
    knn_predict = knn_optimal.predict(X_test)
    print('\nClassification Report:')
    print(metrics.classification_report(y_test, knn_predict))
    
    confusion = metrics.ConfusionMatrixDisplay.from_predictions(y_test, knn_predict)
    confusion.figure_.suptitle('Confusion Matrix')
    plt.show()
